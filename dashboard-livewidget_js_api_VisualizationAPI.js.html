/**
 * @class VisualizationAPI
 * @hideconstructor
 */
define([], () => {
	class VisualizationAPI {

		/**
		 * @description Register a visualization event handler.
         * @function VisualizationAPI#on
		 * @public
		 *
		 * @param {String} name Event name
		 * @param {Function} name Event handler
		 * @return Object An object that has a remove function that removes the event handler.
		 */
		on() {}

		/**
		 * @description Deregister a visualization event handler.
         * @function VisualizationAPI#off
		 * @public
		 *
		 * @param {String} name Event name
		 * @param {Function} name Event handler
		 * 
		 */
		off() {}
		/**
		 * @description Returns the data source used by the visualization.
         * @function VisualizationAPI#getDataSource
		 * @public
		 *
		 * @return DataSourceAPI
		 */
		getDataSource() {}

		/**
		 * @description Set the data source to be used by the visualization.
		 * @function VisualizationAPI#setDataSource
		 * @public
		 *
		 * @param {String} datasourceId Unique data source identifier
		 * @return {Promise<DataSourceAPI>}
		 */
		setDataSource(datasourceId) {
			void(datasourceId);
		}

		/**
		 * @description Get an array of MetadataColumnAPI objects which are being used in the visualization.
		 * @function VisualizationAPI#getUsedMetadataColumnList
		 * @public
		 *
		 * @return {MetadataColumnAPI[]}
		 */
		getUsedMetadataColumnList() {}

		/**
		 * @description Returns the slots interface. The slot interface can be used to get and set the information about the slots and the data mappings in the visualization.
		 * @function VisualizationAPI#getSlots
		 * @public
		 *
		 * @return {SlotsAPI} Slots object
		 *
		 */
		getSlots() {}

		/**
		 * @description Get the data provider interface used to fetch the data to be rendered by the visualization
		 * Supported providers: 'queryService', 'predictService'
		 * Currently, we cannot add a custom dataProvider, but this should be possible through an extension point.
		 *
		 * @function VisualizationAPI#getDataProvider
		 * @public
		 *
		 * @param {String} name
		 *
		 * @returns{Object} Data provider interface. See {@link QueryServiceAPI}, {@link PredictServiceAPI}
		 *
		 */
		getDataProvider(name) {
			void(name);
		}


		/**
		 * @description Return the visulization type. The id indicated the type of visualization instance that is being used.
		 * @function VisualizationAPI#getType
		 * @public
		 *
		 * @returns {String} type  (e.g. Bar, Area, etc)
		 * @example
		 * // will return the string 'Bar'
		 * visualization.getType()
		 *
		 */
		getType() {}


		/**
		 * @description Return true if the widget matches the spec provided in the parameter
		 * We currently only support matches with a spec that has only a  visId and localFilters.
		 * Anything else will cause an error.
		 *
		 * @function VisualizationAPI#matches
		 * @public
		 *
		 * @returns {{visId: String, localFilters: Object}} Spec object to match. For more info about the  content, refer to the visualization specification
		 * @example
		 *
		 * visualization.matches({
		 *  "visId": "com.ibm.vis.rave2bundlecolumn",
		 *  "localFilters": [{
		 *      "id": "banking.Business",
		 *      "columnId": "banking.Business",
		 *      "values": [
		 *          {
		 *              "d": "Corporate Finance",
		 *              "u": "Corporate Finance"
		 *          },
		 *          {
		 *              "d": "Retail Banking",
		 *              "u": "Retail Banking"
		 *          }
		 *      ],
		 *      "excludedValues": [],
		 *      "operator": "notin",
		 *      "type": null,
		 *      "binsLabel": ""
		 *  },
		 *  {
		 *      "id": "banking.Year_",
		 *      "columnId": "banking.Year_",
		 *      "values": [
		 *          {
		 *              "d": "2008",
		 *              "u": "2008"
		 *          },
		 *          {
		 *              "d": "2009",
		 *              "u": "2009"
		 *          }
		 *      ],
		 *      "excludedValues": [],
		 *      "operator": "in",
		 *      "type": null,
		 *      "binsLabel": "Year"
		 *  },
		 *  {
		 *      "id": "banking.Recovery_Amount_sum",
		 *      "columnId": "banking.Recovery_Amount",
		 *      "values": [
		 *          {
		 *              "d": 31234327.54830128,
		 *              "u": 31234327.54830128
		 *          },
		 *          {
		 *              "d": 35389734.47129045,
		 *              "u": 35389734.47129045
		 *          }
		 *      ],
		 *      "excludedValues": [],
		 *      "operator": "between",
		 *      "aggregationType": "sum",
		 *      "preOrPost": "post"
		 *  }]
		 *})
		 *
		 */
		matches() {}


		/**
		 * @description Return the visualization definition. The definition describes various aspects of the visualization like name, slots, properties, etc...
		 * @function VisualizationAPI#getDefinition
		 * @public
		 *
		 * @returns {VisDefinitionAPI} vis defintion
		 *
		 */
		getDefinition() {}

		/**
		 * @function Visualization#getSavedPrompts
		 * @description Get the API of SavedPrompts
		 * @public
		 *
		 * @return {SavedPromptsAPI} API of SavedPrompts
		 */
		getSavedPrompts() {}

		/**
		 * @description Get the local filters.
		 * @function VisualizationAPI#getLocalFilters
		 * @public
		 *
		 * @returns {FiltersAPI}
		 */
		getLocalFilters() {}

		/**
		 * @description Set the visualization type. Type 'auto' will set the visualization to the recommended type.
		 * @function VisualizationAPI#setType
		 * @public
		 *
		 * @param {String} type (e.g. auto, Bar, Area, etc)
		 * @param {SetOptions} options
		 *
		 * @todo - define the type enum (can be found in SmartsToVizMapping.js)
		 */
		setType(type, options) {
			void(type);
			void(options);
		}

		/**
		 *  @description Indicate whether the visualization type is locked or not.
		 *  An unlocked visualization type will use the recommender when new columns are added and the viz type might change as we add more columns.
		 *  A locked visualization type will always used the current visualization as we add new columns to the slots.
		 *
		 * @function VisualizationAPI#isTypeLocked
		 * @public
		 *
		 * @param {Boolean} isTypeLocked
		 *
		 */
		isTypeLocked() {}

		/**
		 *  @description Lock the visualization.
		 *  An unlocked visualization type will use the recommender when new columns are added and the viz type might change as we add more columns.
		 *  A locked visualization type will always used the current visualization as we add new columns to the slots.
		 *
		 * @function VisualizationAPI#lockType
		 * @public
		 *
		 * @param {SetOptions} Options
		 *
		 */
		lockType(options) {
			void(options);
		}


		/**
		 * @description Unlock the visualization.
		 * An unlocked visualization type will use the recommender when new columns are added and the viz type might change as we add more columns.
		 * A locked visualization type will always used the current visualization as we add new columns to the slots.
		 * @function VisualizationAPI#unlockType
		 * @public
		 *
		 * @param {SetOptions} Options
		 *
		 */
		unlockType(options) {
			void(options);
		}


		/**
		 * @description Returns the decorator object that can be used to decorate the data in the visualization.
		 * If the visualizatoin support multiple datasets and no view id is provided, then the decorator for the first dataset is returned.
		 *
		 * @function VisualizationAPI#getDecorator
		 * @public
		 *
		 * @param {String} [viewId]  Data view id. If not provided, the decorator for the first data view is returned.
		 *
		 * @return {DecoratorAPI} decorator object
		 *
		 */
		getDecorator(dataViewId) {
			void(dataViewId);
		}


		/**
		 * @description Return a list of decorator object that can be used to decorate the data in the visualization.
		 * Each dataset will have its own decorator object.
		 *
		 * @function VisualizationAPI#getDecoratorList.
		 * @public
		 *
		 * @return {DecoratorAPI[]} decorator object
		 *
		 */
		getDecoratorList() {}


		/**
		 * @description Returns true if the hasUnavailableMetadataColumns references a column that is no longer available
		 * @function VisualizationAPI#getDecoratorList
		 * @public
		 *
		 * @returns {Boolean}
		 */
		hasUnavailableMetadataColumns() {}

		/**
		 * @description Returns an array of slot actions for the slot identify by the data slot id and and itemIndices
		 * @function VisualizationAPI#getSlotActionList
		 * @public
		 *
		 * @param {string} dataSlotId The data slot id
		 * @param {number[]|number} [itemIndices] The indices to the data item(s) in the slot
		 * @param {object} [options] Options to retrieve the slot action list
		 * @returns {array} Contains an array of slot actions
		 */
		getSlotActionList(dataSlotId, itemIndices, options) {
			void(dataSlotId);
			void(itemIndices);
			void(options);
		}
	}

	/**
	 * @description Enum for all supported DECORATIONS
	 * @readonly
	 * @enum {String}
	 * @constant
	 * @memberOf VisualizationAPI
	 */
	VisualizationAPI.DECORATIONS = {
		HIGHLIGHT: 'highlight',
		SELECT: 'select',
		TOOLTIP: 'tooltip'
	};

	return VisualizationAPI;
});
