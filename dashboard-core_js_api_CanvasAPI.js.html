/**
 * @class CanvasAPI
 * @hideconstructor
 * @classdesc API class that is used to control a dashboard canvas behaviour.
 */
define([], () => {
	class CanvasAPI {

		/**
		 * @description Register a canvas event handler.
		 * @function CanvasAPI#on
		 * @public
		 *
		 * @param {String} name Event name
		 * @param {Function} name Event handler
		 * @return Object An object that has a remove function that removes the event handler.
		 */
		on() {}

		/**
		 * @description Deregister a canvas event handler.
		 * @function CanvasAPI#off
		 * @public
		 *
		 * @param {String} name Event name
		 * @param {Function} name Event handler
		 *
		 */
		off() {}

		/**
		 * @function CanvasAPI#addContent
		 * @description Add content to the canvas
		 * @param {Object} content Content used to add a widget to the canvas
		 * @param {String} content.containerId Container id where to add the content. If not provided, an automatic position will be decided.
		 * @param {Number} content.position Index where to add the content. If not provided, an automatic position will be decided.
		 * @param {Object} content.properties An object of property names & values to apply to the content
		 * @param {String} content.type Content provider type to use. Must provide a spec if not defined.
		 * @param {Object} content.spec Layout, widget or fragment spec.
		 * @param {TransactionToken} transactionToken
		 * @return {Promise<ContentAPI>} new content
		 */
		addContent() {}

		/**
		 * @function CanvasAPI#moveContent
		 * @description move the specified contents @contentIdList to new container @containerId
		 * @param {String} containerId
		 * @param {*} contentIdList
		 * @param {*} transactionToken
		 * @return {[contentApi]} the list of contentAPIs correnponding to @contentIdList
		 */
		moveContent() {}

		/**
		 * Return the content that match the given id
		 * @function CanvasAPI#getContent
		 * @param {String} id - The ID of the content
		 * @return {ContentAPI}
		 */
		getContent() {}


		/**
		 * Return a list of toolbar actions
		 *
		 * @param {String[]} idList
		 * @return {Object[]} array of actions
		 * @example
		 *  {
				name: 'group',
				label: stringResources.get('toolbarActionGroup'),
				icon: 'dashboard-group',
				type: 'Button',
				action: () => {}
			}
		 */
		getContentActionList(idList) {
			void(idList);
		}

		/**
		 * Select the contents that match the given ids
		 * @function CanvasAPI#selectContent
		 * @param {String[]} idList- Array if ids to select
		 */
		selectContent() {}

		/**
		 * Deselect the contents that match the given ids
		 * @function CanvasAPI#deselectContent
		 * @param {String[]} idList- Array if ids to select
		 */
		deselectContent() {}

		/**
		 * Get the list of content that is selected
		 * @function CanvasAPI#getSelectedContentList
		 * @param {Object} selector - Optional, returns groups and widgets by default.
		 * @param {String} selector.type - selector string
		 * @return {ContentAPI[]} contentList  - Array of content
		 */
		getSelectedContentList() {}


		/**
		 * Remove the content with the given id
		 * @function CanvasAPI#deleteContent
		 * @param {String} id - The ID of the content
		 *
		 * @param {TransactionToken} transactionToken
		 */
		removeContent() {}

		/**
		 * Return a list of content that match the provided selector
		 * @function CanvasAPI#findContent
		 * @param {Object} selector - selector object
		 * @param {String} selector.type - selector string
		 * @return {ContentAPI[]} Array of contents
		 */
		findContent() {}

		/**
		 * @function DashboardAPI#copy
		 * @description Copy the currently selected widget(s) on the dashboard
		 */
		copy() {}

		/**
		 * @function DashboardAPI#paste
		 * @description Paste the copied widget(s) to the dashboard
		 * @return {Promise} promise which is resolved once the widget(s) are successfully pasted
		 */
		paste() {}

		/**
		 * @function CanvasAPI#getPropertyLayoutList
		 * @descrpition Returns a list of layout and properties used to render the properties UI
		 * @return {Object[]} Array of layout and properties used to render the properties UI
		 */
		getPropertyLayoutList() {}

		/**
		 * @function CanvasAPI#getPropertiesNameList
		 * @description Returns an object of property objects
		 * @return {String[]} Array of property objects
		 */
		getPropertyNameList() {}

		/**
		 * @function CanvasAPI#setPropertyValue
		 * @description Sets the value for a property
		 * @param {String} name - The name of the property to set
		 * @param {Object} value - The value to set
		 * @param {TransactionToken} transactionToken
		 */
		setPropertyValue() {}

		/**
		 * @function CanvasAPI#getPropertyValue
		 * @description Get the value for a property
		 * @param {String} name - The name of the property
		 * @return {Object} The value of the property
		 */
		getPropertyValue() {}
	}

	return CanvasAPI;
});
