/**
 * @class DataItemAPI
 * @hideconstructor
 * @classdesc implementation of the DataItemAPI interface.
 */
define([], () => {
	class DataItemAPI {

		/**
		 * @description Return the unique id that identifies the given dataitem.
		 *
		 * @function DataItemAPI#getId
		 * @public
		 *
		 * @return {String} id
		 */
		getId() { }

		/**
		 * @description Return the dataitem label.
		 *
		 * @function DataItemAPI#getLabel
		 * @public
		 *
		 * @return {String} label
		 */
		getLabel() { }

		/**
		 * @description Return the item id that this data item reference. An item id is the metadata column id.
		 *
		 * @function DataItemAPI#getColumnId
		 * @public
		 *
		 * @return {String} columnId
		 */
		getColumnId() { }



		/**
		 * @description Return the item data type.
		 * @function DataItemAPI#getDataType
		 * @public
		 *
		 * @return {String} dataType { 'string' | 'decimal' }
		 */
		getDataType() { }

		/**
		 * @description Return trun if the data item is a data or time type. Data or time types include: date, datetime, time, year.
		 * @function DataItemAPI#isDateOrTimeType
		 * @public
		 *
		 * @return {Boolean} isDateOrTimeType
		 */
		isDateOrTimeType() { }

		/**
		 * @description Return the type of the data item. The type can be either 'attribute' or 'fact'.
		 * @function DataItemAPI#getType
		 * @public
		 *
		 * @return {String} type
		 */
		getType() { }

		/**
		 * @description Return the dataitem aggregation type.
		 * @function DataItemAPI#getAggregation
		 * @public
		 *
		 * @return {String} aggregationType
		 *
		 */
		getAggregation() { }


		/**
		 * @description Set the dataitem aggregation type.
		 * @function DataItemAPI#setAggregation
		 * @public
		 *
		 * @param {aggregation} aggregation
		 * @param {SetOptions} options
		 */
		setAggregation(aggregation, options) {
			void (aggregation);
			void (options);
		}

		/**
		 * @description Return true if the dataitem aggregation type was not explicitly set.
		 * @function DataItemAPI#hasDefaultAggregation
		 * @public
		 *
		 * @return {Boolean} hasDefaultAggregation
		 *
		 */
		hasDefaultAggregation() { }


		/**
		 * @description Return true if the dataitem sort type was not explicitly set.
		 * @function DataItemAPI#hasDefaultSort
		 * @public
		 *
		 * @return {Boolean} hasDefaultSort
		 *
		 */
		hasDefaultSort() { }

		/**
		 * @description Return true if the dataitem format type was not explicitly set.
		 * @function DataItemAPI#hasDefaultFormat
		 * @public
		 *
		 * @return {Boolean} hasDefaultFormat
		 *
		 */
		hasDefaultFormat() { }

		/**
		 * @description Return the column metadata that is associated with this data item.
		 *
		 * @function DataItemAPI#getMetadataColumn
		 * @public
		 *
		 * @return {MetadataColumnAPI} columnMetadata Return null if the metadata item is not available
		 */
		getMetadataColumn() { }


		/**
		 * @description Return the data format for this data item.
		 * @function DataItemAPI#getFormat
		 * @public
		 *
		 * @param {Object} options ignoreDefaultFormatting - Ignore any default formatting
		 * @return {FormatSpec} return format spec
		 */
		getFormat(options) {
			void (options);
		}


		/**
		 * @description Set the format spec for the given data item.
		 * @function DataItemAPI#setFormat
		 * @public
		 *
		 * @param {FormatSpec} formatSpec
		 * @param {SetOptions} options
		 *
		 */
		setFormat(formatSpec, options) {
			void (formatSpec);
			void (options);
		}

		/**
		 * @description Get the effective sort spec. Is there is a user sort set, then it is returned, otherwise, the default sort is returned.
		 *
		 * @function DataItemAPI#getSort
		 * @public
		 * @return {SortSpec}
		 *
		 * @example
		 *  {
		 *    type: String ('asc' | 'desc' | 'null'),
		 *    priority: Number,
		 *    by: String  ('caption' | 'value'),
		 *    context: String (column id used for the sort context)
		 *    custom: Object (specify sort overrides)
		 *    custom.top: String[] (Use values of members to move to the top of the sorted list)
		 *    custom.enabled: Boolean (indicate if custom sort is enabled)
		 *  }
		 *
		 */
		getSort() { }

		/**
		 * @description Get the top/bottom information that is set on the data item.
		 * @function DataItemAPI#getTopBottom
		 * @public
		 * @example
		 *  {
		 *    type: String ('topcount' | 'toppercent' | 'bottomcount' | 'bottompercent'),
		 *    value: Number,
		 *    rank: boolean
		 *    context: Object (with itemId and aggregate)
		 *  }
		 *
		 * @return {TopBottomInfo}
		 *
		 */
		getTopBottom() { }

		/**
		 * @description Get the top/bottom information that is set on the data item.
		 * @function DataItemAPI#setTopBottom
		 * @public
		 *
		 * @example
		 *  {
		 *		type: String ('topcount' | 'toppercent' | 'bottomcount' | 'bottompercent'),
		 *		value: Number,
		 *		rank: boolean
		 *    context:  the measure (column id) can be used when the topbottom applied on the categories
		 *  }
		 *
		 * @param {TopBottomInfo} topBottomInfo
		 * @param {SetOptions} options
		 *
		 */
		setTopBottom(topBottomInfo, options) {
			void (topBottomInfo);
			void (options);
		}

		/**
		 * @description Return true if the data item is a hierarchy and we can drill down.
		 * @function DataItemAPI#canDrillDown
		 * @public
		 *
		 * @return {Boolean} canDrillDown
		 */
		canDrillDown() { }


		/**
		 * @description Drill down.
		 *
		 * @function DataItemAPI#drillDown
		 * @public
		 *
		 * @param {String} value
		 * @param {SetOptions} options
		 *
		 *
		 */
		drillDown(value, options) {
			void (value);
			void (options);
		}

		/**
		 * @description Get the Drill down value (member id) if a drill down is set.
		 *
		 * @function DataItemAPI#getDrillDownValue
		 * @public
		 *
		 * @return {String} drillDown value
		 */
		getDrillDownValue() {}


		/**
		 * @description Drill up.
		 *
		 * @function DataItemAPI#drillUp
		 * @public
		 *
		 * @param {String} value
		 * @param {SetOptions} options
		 *
		 *
		 */
		drillUp(value, options) {
			void (value);
			void (options);
		}

		/**
		 * @description Get the Drill up value (member id) if a drill up is set.
		 *
		 * @function DataItemAPI#getDrillDownValue
		 * @public
		 *
		 * @return {String} drillUp value
		 */
		getDrillUpValue() {}


		/**
		 * @description Clear the drill up or down selection
		 *
		 * @function DataItemAPI#clearDrill
		 * @public
		 *
		 * @param {SetOptions} options
		 */
		clearDrill() {}

		/**
		 * @description Enable binning in the data item.
		 * @function DataItemAPI#setBinning
		 * @public
		 *
		 * @param {BinningSpec} binningSpec - null to clear the binning
		 * @param {SetOptions} options
		 *
		 */
		setBinning(binningSpec, options) {
			void (binningSpec);
			void (options);
		}


		/**
		 * @description Get the binning json object if specified for this data item.
		 * @function DataItemAPI#getBinning
		 * @public
		 *
		 * @return {BinningSpec}
		 */
		getBinning() { }



		/**
		 * @description Set the type, sort-by method and priority of sort on the data item.
		 *
		 * @function DataItemAPI#setSort
		 * @public
		 * @param {SortSpec} sortSpec
		 * @param {SetOptions} options
		 *
		 */
		setSort(sortSpec, options) {
			void (sortSpec);
			void (options);
		}
	}

	/**
	 * @description Enum for all supported topbottom type.
	 * @readonly
	 * @enum {String}
	 * @constant
	 * @memberOf DataItemAPI
	 */
	DataItemAPI.TOPBOTTOM_TYPE = {
		topcount : 'topcount',
		toppercent : 'toppercent',
		bottomcount : 'bottomcount',
		bottompercent : 'bottompercent'
	};
	return DataItemAPI;
});
