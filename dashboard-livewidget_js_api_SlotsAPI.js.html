/**
 * @class SlotsAPI
 * @hideconstructor
 * @classdesc implementation of the SlotsAPI interface
 */
define([], () => {
	class SlotsAPI {

		/**
		 * @function SlotsAPI#addColumns
		 * @description Add columns to the visualization. The recommender will decide to which slots the columns are assigned.
		 * @public
		 *
		 * @param {string[]} itemIds an array of column item identifiers
		 * @param {SetOptions} Options
		 */
		addColumns(itemIds, options) {
			void(itemIds);
			void(options);
		}

		/**
		 * @function SlotsAPI#setDataItems
		 * @description Make changes to the dataItems assigned to an existing slot. Supports reorder and delete existing items.
		 * @public
		 *
		 * @param {string[]} dataItemtIds - an ordered array of dataItem IDs to assign to this slot.
		 * @param {string} slotId
		 * @param {SetOptions} options
		 * @return {DataItemAPI[]} An array of data items that we removed
		 */
		setDataItems(dataItemIds, slotId, options) {
			void(dataItemIds);
			void(slotId);
			void(options);
		}

		/**
		 * @function SlotsAPI#swapSlots
		 * @description Swap data items between the two slots that are identified by their IDs
		 * the operation should be one transaction
		 * @public
		 *
		 * @param {string} sourceSlotId  The ID of the source slot
		 * @param {string} targetSlotId  The ID of the target slot
		 */
		swapSlots(sourceSlotId, targetSlotId) {
			void(sourceSlotId);
			void(targetSlotId);
		}

		/**
		 * @function SlotsAPI#getSlot
		 * @description Get the slot with the given id
		 * @public
		 *
		 * @param {string} slotId
		 * @return {SlotAPI} slot
		 */
		getSlot(slotId) {
			void(slotId);
		}

		/**
		 * @function SlotsAPI#getSlotList
		 * @description Return the list of all the slots
		 * @public
		 *
		 * @return {SlotAPI[]}
		 */
		getSlotList() {}

		/**
		 * @function SlotsAPI#getMappedSlotList
		 * @description Return the list of the mapped slots
		 * @public
		 *
		 * @return {SlotAPI[]}
		 */
		getMappedSlotList() {}

		/**
		 * @function SlotsAPI#getDataItemList
		 * @description Return a list of all the dataitem mapped to all the slots
		 * @public
		 *
		 * @return {DataItem[]} dataItemList
		 */
		getDataItemList() {}

		/**
		 * @function SlotsAPI#getMappingInfo
		 * @description Return the data item location in the slots for the supplied dataItemUniqueId.
		 * @public
		 *
		 * @param {string} dataItemId - the dataItemUniqueId to find in the set of slots.
		 * @return {MappingInfo}  the dataItem's slot location information (dataItemAPI, slotAPI, indexInSlot). Returns null if the data item is not mapped.
		 */
		getMappingInfo(dataItemId) {
			void(dataItemId);
		}

		/**
		 * @function SlotsAPI#getMappingInfoList
		 * @description Return an array of all the data item mapping info for all the slots.
		 * @public
		 *
		 * @return {MappingInfo[]} An array of data item mapping info
		 */
		getMappingInfoList() {}

		/**
		 * @function SlotsAPI#isMappingComplete
		 * @description Returns true if all of the required slots are mapped and this visualization can be rendered.
		 * @public
		 *
		 * @return {boolean} isMappingComplete
		 */
		isMappingComplete() {}

		/**
		 * @function SlotsAPI#clearAllSlots
		 * @description Clear all slots content
		 * @public
		 *
		 * @param {SetOptions} options
		 */
		clearAllSlots(options) {
			void(options);
		}
	}

	return SlotsAPI;
});
