define([], () => {
	class ConditionalFormattingAPI {

		isEnabled() {}

		initializeFeature() {}

		 * @return {Promise} Promise that's resolved with an Object containing a color and shapeId property.

		getFormatting() {}

		getShowColors() {}

		 * @function ConditionalFormattingAPI#setShowColors
		 * @description Set whether or not conditional formatting should be applied
		 * @param {Boolean} enabled - whether or not conditional formatting is enabled

		setShowColors() {}

		/**
		 * @function ConditionalFormattingAPI#getPaletteId
		 * @description Get the current palette Id for conditional formatting
		 * @return {String}
		 */
		getPaletteId() {}

		 * @function ConditionalFormattingAPI#setPaletteId
		 * @description Sets the conditional palette id
		 * @param {String} paletteId - the new palette id
		 * @return {Promise}
		 */
		setPaletteId() {}

		/**
		 * @function ConditionalFormattingAPI#getPaletteReversed
		 * @description Returns true if the conditional palette should be reversed.
		 * @return {Boolean}
		 */
		getPaletteReversed() {}

		/**
		 * @function ConditionalFormattingAPI#setPaletteReversed
		 * @description Sets if the conditional palette should be reversed
		 * @param {Boolean} reversed
		 */
		setPaletteReversed() {}

		/**
		 * @function ConditionalFormattingAPI#getPropertyLayoutList
		 * @description get list of property layouts
		 * @return {Object[]}
		 */
		getPropertyLayoutList() {}

		/**
		 * @function ConditionalFormattingAPI#getPropertyList
		 * @description get the list of properties
		 * @return {Object[]}
		 */
		getPropertyList() {}

		/**
		 * @function ConditionalFormattingAPI#getRangeScale
		 * @description Returns the type of range scale (% or #)
		 * @returns {String} Returns the type of range scale (% or #)
		 */
		getRangeScale() {}

		/**
		 * @function ConditionalFormattingAPI#setRangeScale
		 * @description Sets the range scale to be percentage (%) or numeric (#)
		 * @param {String} rangeScale - Must be '%' or '#'
		 */
		setRangeScale() {}

		/**
		 * @function ConditionalFormattingAPI#getNumberOfRanges
		 * @description Returns the number of ranges
		 * @return {Integer} number of ranges
		 */
		getNumberOfRanges() {}

		 * @param {Integer} numberOfRanges - Number of ranges to have.  Must be higher then zero.
		 */
		setNumberOfRanges() {}

		/**
		 * @function ConditionalFormattingAPI#removeRange
		 * @description Removes a specific range
		 * @param {Integer} rangeIndexToRemove - The index of the range to be removed
		 */
		removeRange() {}

		/**
		 * @function ConditionalFormattingAPI#getRanges
		 * @description Returns an array of range objects
		 * @return {[Ranges]}

		getRanges() {}

		 * @function ConditionalFormattingAPI#setStartValueForRange
		 * @description Sets the starting value for a specific range
		 * @param {Integer} startValue - The value to set as the starting value.  Pass undefined if you want to set the start value to 'min'
		 * @param {Integer} rangeIndex - The index of the range you want to update
		 * @return {SetValueResponse}
		 */
		setStartValueForRange() {}

		/**

		 * @return {SetValueResponse}
		 */
		setEndValueForRange() {}

		/**

		 */
		setColorIdForRange() {}

		/**

		 * @return {String} - the hex color
		 */
		getColorHexFromRange() {}

		/**
		 * @function ConditionalFormattingAPI#getColorIdFromRange
		 * @description Get the internal color id for a specific range
		 * @param {Integer} rangeIndex - The index of the range you want to get the colorId from
		 * @return {String} - the color id
		 */
		getColorIdFromRange() {}
	}
	return ConditionalFormattingAPI;
